void strcpyimpl(char* dst, char* src)//方法一
{
	while (*src!=0)
	{
		*dst = *src;
		src++;
		dst++;
	}
	//补结束符，因为strlen获取的数据不包含结束符
	*dst = 0;
	//dst[strlen(src)] = 0;
}
